MRound 

CREATE FUNCTION dbo.mround(@my_val numeric, @my_multiple numeric)
RETURNS numeric
BEGIN;
    --==================================================
    --Author:     Daniel Van Den Bosch, VanData LLC
    --Created At: 11/29/2021
    --==================================================
   DECLARE @retval numeric;
   SELECT @retval = Round(@my_val / @my_multiple, 0) * @my_multiple
   RETURN @retval;
END
go
